\subsection{fibonacci.c3p}

\begin{verbatim}
func fibonacci(n : i32)
    if (n > 1)
        f1 : i32 = call fibonacci n - 1
        f2 : i32 = call fibonacci n - 2
        ret f1 + f2
    endif

    ret n
endfunc

proc main()
    n10 : i32 = call fibonacci 10
    n20 : i32 = call fibonacci 20
    n30 : i32 = call fibonacci 30
